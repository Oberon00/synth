set(libsynth_HDRS
    "CgStr.hpp"
    "FileIdSupport.hpp"
    "MultiTuProcessor.hpp"
    "SimpleTemplate.hpp"
    "cgWrappers.hpp"
    "config.hpp"
    "output.hpp"
)

set(libsynth_SRCS
    "MultiTuProcessor.cpp"
    "SimpleTemplate.cpp"
    "annotate.cpp"
    "debug.cpp"
    "output.cpp"

)
